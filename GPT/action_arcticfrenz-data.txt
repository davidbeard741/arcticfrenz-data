          content:
            application/json:
              schema:
                type: array
                items:
                  type: object
                  properties:
                    holderAddress:
                      type: string
                    quantityNfts:
                      type: integer
                    holdingNfts:
                      type: array
                      items:
                        type: object
                        properties:
                          nftAddress:
                            type: string
                          rarityScore:
                            type: number
                          daysHeld:
                            type: integer
                    holderScore:
                      type: number
